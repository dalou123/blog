    # ES新特性实践应用

    # 可选链操作符

    告别繁琐的空值检查，用简单的 ?. 优雅处理对象属性访问。

    // 之前的写法
const street = user && user.address && user.address.street;

// 现在的写法
const street = user?.address?.street;

    # 空值合并运算符

    使用 ?? 来处理 null 或 undefined 的默认值设置。

    const value = null;
const defaultValue = value ?? 'default';  // 'default'

    # 私有类字段

    使用 # 声明私有字段，增强面向对象编程的封装性。

    class Person {
  #name;
  constructor(name) {
    this.#name = name;
  }
  
  getName() {
    return this.#name;
  }
}

    简化条件赋值操作

// 空值合并赋值
